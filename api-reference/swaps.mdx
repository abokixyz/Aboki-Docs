Create a new swap between Nigerian Naira (NGN) and USDC.

<ParamField header="Authorization" type="string" required>
  Bearer token for API authentication
</ParamField>

<ParamField body="fromCurrency" type="string" required>
  Source currency. Must be either `NGN` or `USDC`
</ParamField>

<ParamField body="toCurrency" type="string" required>
  Target currency. Must be either `USDC` or `NGN`
</ParamField>

<ParamField body="amount" type="number" required>
  Amount to swap (in the source currency)
</ParamField>

<ParamField body="userWallet" type="string" required>
  User's crypto wallet address for USDC transactions
</ParamField>

<ParamField body="bankAccount" type="string">
  Nigerian bank account number (required when toCurrency is NGN)
</ParamField>

## Response

<ResponseField name="data" type="object">
  <ResponseField name="swapId" type="string">
    Unique identifier for this swap transaction
  </ResponseField>
  <ResponseField name="status" type="string">
    Current status: pending, processing, completed, failed
  </ResponseField>
  <ResponseField name="fromAmount" type="number">
    Amount being swapped from
  </ResponseField>
  <ResponseField name="toAmount" type="number">
    Amount being swapped to
  </ResponseField>
</ResponseField>
