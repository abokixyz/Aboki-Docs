
```mdx
---
title: "Create Swap"
api: "POST https://api.aboki.xyz/v1/swaps"
description: "Create a new fiat ↔ crypto swap transaction"
---

Create a new swap between Nigerian Naira (NGN) and USDC.

<ParamField header="Authorization" type="string" required>
  Bearer token for API authentication
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`
</ParamField>

## Body Parameters

<ParamField body="fromCurrency" type="string" required>
  Source currency. Must be either `NGN` or `USDC`
</ParamField>

<ParamField body="toCurrency" type="string" required>
  Target currency. Must be either `USDC` or `NGN`
</ParamField>

<ParamField body="amount" type="number" required>
  Amount to swap (in the source currency)
</ParamField>

<ParamField body="userWallet" type="string" required>
  User's crypto wallet address for USDC transactions
</ParamField>

<ParamField body="bankAccount" type="string">
  Nigerian bank account number (required when toCurrency is NGN)
</ParamField>

<ParamField body="callbackUrl" type="string">
  Webhook URL to receive status updates
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Swap Details">
    <ResponseField name="swapId" type="string">
      Unique identifier for this swap transaction
    </ResponseField>
    <ResponseField name="status" type="string">
      Current status: `pending`, `processing`, `completed`, `failed`
    </ResponseField>
    <ResponseField name="fromCurrency" type="string">
      Source currency (NGN or USDC)
    </ResponseField>
    <ResponseField name="toCurrency" type="string">
      Target currency (USDC or NGN)
    </ResponseField>
    <ResponseField name="fromAmount" type="number">
      Amount being swapped from
    </ResponseField>
    <ResponseField name="toAmount" type="number">
      Amount being swapped to
    </ResponseField>
    <ResponseField name="exchangeRate" type="number">
      Applied exchange rate
    </ResponseField>
    <ResponseField name="fees" type="object">
      <Expandable title="Fee Structure">
        <ResponseField name="abokiFee" type="number">
          Aboki platform fee
        </ResponseField>
        <ResponseField name="networkFee" type="number">
          Blockchain network fee
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="estimatedCompletion" type="string">
      ISO timestamp of estimated completion
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO timestamp of swap creation
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable success message
</ResponseField>

<ResponseField name="statusCode" type="number">
  HTTP status code
</ResponseField>
