# Get Offramp Quote

Calculate how much NGN your customer will receive for selling crypto tokens. Shows exact amounts including fees and exchange rates.

<ParamField header="Authorization" type="string" required>
  Bearer token from your business login
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be application/json
</ParamField>

## Body Parameters

<ParamField body="tokenAmount" type="number" required>
  Amount of tokens to convert (e.g., 100 for 100 USDC)
</ParamField>

<ParamField body="targetToken" type="string" required>
  Token symbol to convert: USDC, USDT, ETH, SOL
</ParamField>

<ParamField body="targetNetwork" type="string" required>
  Blockchain network: base, ethereum, or solana
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Whether the quote was generated successfully
</ResponseField>

<ResponseField name="data" type="object">
  <ResponseField name="tokenAmount" type="number">
    Amount of tokens being converted
  </ResponseField>
  <ResponseField name="grossNgnAmount" type="number">
    Total NGN value before fees
  </ResponseField>
  <ResponseField name="feeAmount" type="number">
    Fee amount in NGN
  </ResponseField>
  <ResponseField name="netNgnAmount" type="number">
    Final NGN amount customer receives
  </ResponseField>
  <ResponseField name="exchangeRate" type="number">
    Current USD to NGN exchange rate
  </ResponseField>
  <ResponseField name="expiresAt" type="string">
    Quote expiration time (5 minutes)
  </ResponseField>
  <ResponseField name="breakdown" type="object">
    Detailed breakdown for customer display
  </ResponseField>
</ResponseField>
